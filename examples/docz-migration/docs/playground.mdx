## Stateful

<Playground>
  {() => {
    const [counter, setCounter] = React.useState(0);
    return (
      <>
        <Button onClick={() => setCounter(counter => counter+1)}>
          Increase
        </Button>
        <p>{counter}</p>
      </>
    )
  }}
</Playground>
