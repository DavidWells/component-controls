import { Fragment, useEffect, useState } from 'react';
import * as themes from '@theme-ui/presets';
import { ControlTypes, defaultStore } from '@component-controls/core';
import { useStore, useThemeState } from '@component-controls/store';
import { AppContext } from '@component-controls/app';
import { Source } from '@component-controls/components';
import { Playground, Story, PropsTable } from '@component-controls/blocks';


component-controls allows full customization of the user interface - from theming, SEO tags to toolbars, sidebars and footer. Most of the user-interface customizations can be set up in your .`config/runtime.js` file.

## Installing a theme

You can use any [theme-ui](https://theme-ui.com) theme as a starting point for customizing the user interface.

<Story name="select-theme" controls={{ theme: { type: ControlTypes.OPTIONS, options: ['none', ...Object.keys(themes)], value: 'none' }}}>
  {({ theme: themeName }) => {
    const store = useStore();
    const [globalTheme, setGlobalTheme] = useThemeState();
    const theme = themes[themeName] || store.config.theme;
    useEffect(() => {
      if (globalTheme !== theme) {
        setGlobalTheme(theme)
      }
    }, [theme]);  
    return (
        <Source title=".config/runtime.js">
{`${themeName !== 'none' ? `import { ${themeName} } from '@theme-ui/presets';` : ''}
export default {
  theme: ${themeName !== 'none' ? themeName : '{}'},
}`}
      </Source>
    );  
  }}
</Story>
